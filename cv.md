### Code example:
*Write a bubble sort algorithm in JavaScript*
```
function bubbleSort (arr) {
    for (let i = 0, i < arr.length - 1; i++) {
        for (let j = 0, j < arr.length - i; j++) {
            if (arr[j] > arr[j + 1]) {
                let swap = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = swap;
            }
        }
    }
return arr;
}
```
